function combine(image1, image2){
    var newImg = new SimpleImage(image1.getWidth(), image1.getHeight());
    for(var px of image1.values()){
        var x = px.getX();
        var y = px.getY();
        var px2 = image2.getPixel(x, y)
        var imR = newpv(px.getRed(), px2.getRed());
        var imG = newpv(px.getGreen(), px2.getGreen());
        var imB = newpv(px.getBlue(), px2.getBlue());

function extract(image){
	var output = new SimpleImage(image.getWidth(), image.getHeight());
	for(var px of image.values()){
		var x = px.getX();
		var y = px.getY();
		var opx = output.getPixel(x, y);
		var inRed = px.getRed();
		opx.setRed(inRed - (Math.floor(inRed/8)*8));
		var inGreen = px.getGreen();
		opx.setGreen(inGreen - (Math.floor(inGreen/8)*8));
		var inBlue = px.getBlue();
		opx.setBlue(inBlue-(Math.floor(inBlue/8)*8));
		output.setPixel(x, y, opx);
	}
	return output;
}
