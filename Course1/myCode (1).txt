var image = new SimpleImage(200,200);
//print(image);

for(var px of image.values()){
    if(px.getY()<=(200/3)){
        px.setRed(255);
    }
    else if(px.getY()>(400/3)){
        px.setBlue(255);
    }
    else if(px.getY()>(200/3) && px.getY()<=(400/3) && px.getX()>(200/3)){
        px.setGreen(255);
    }
    else if(px.getY()>(200/3) && px.getX()<(200/3)) {
        px.setRed(255);
    }
    
}
print(image);
